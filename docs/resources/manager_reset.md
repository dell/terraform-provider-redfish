resource "redfish_manager_reset" "manager_reset" {
  for_each = var.rack1

  redfish_server {
    user         = each.value.user
    password     = each.value.password
    endpoint     = each.value.endpoint
    ssl_insecure = each.value.ssl_insecure
  }

  id         = "iDRAC.Embedded.1"
  reset_type = "GracefulRestart"
}
