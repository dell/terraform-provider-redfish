  for_each = var.rack1

  redfish_server {
    user          = each.value.user
    password      = each.value.password
    endpoint      = each.value.endpoint
    validate_cert = each.value.validate_cert
  }
}

output "idrac_attributes" {
  value     = data.redfish_dell_idrac_attributes.idrac
  sensitive = true
}
